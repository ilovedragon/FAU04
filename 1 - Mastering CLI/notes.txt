mkdir: Make directory
mkdir [directory_name]

rmdir: Remove directory
rmdir [directory_name]

pwd: Print/Prsent working directory
pwd

ls: List Command
ls

cd: change directory
cd [directory_path]

cd ..: Go back one directory.
cd ..

touch; Create files
touch [file_name]

rm: Remove files
rm [file_name]

cp: Copy files
cp [source_file] [destination_path]

mv: Move file
mv [source_file] [destination_path]
